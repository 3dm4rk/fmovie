// Use File System Access API async function selectVideosFolder() { try { const dirHandle = await window.showDirectoryPicker(); // Scan recursively for video files const files = await scanDirectory(dirHandle); // Build movie list from files // ... render } catch (err) { // Fallback to virtual file system } } async function scanDirectory(dirHandle, path = '') { const entries = []; for await (const entry of dirHandle.values()) { if (entry.kind === 'file') { const ext = entry.name.split('.').pop().toLowerCase(); if (['mp4', 'mkv', 'avi', 'mov', 'webm'].includes(ext)) { entries.push({ name: entry.name, handle: entry, path }); } } else if (entry.kind === 'directory') { const subEntries = await scanDirectory(entry, path + '/' + entry.name); entries.push(...subEntries); } } return entries; }
